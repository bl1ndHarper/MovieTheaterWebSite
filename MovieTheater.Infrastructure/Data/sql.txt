DO $$
DECLARE
    standard_sector_id BIGINT;
    vip_sector_id BIGINT;
    row_letter CHAR;
    col INTEGER;
    i INTEGER;
BEGIN
    -- Отримуємо ID секторів для Red Hall
    -- Використовуємо явні Id секторів, які ми щойно вставили
    standard_sector_id := 1;
    vip_sector_id := 2;

    -- Звичайні ряди: A–J (10 рядів × 12 місць = 120 сидінь)
    FOR i IN 65..74 LOOP  -- ASCII 'A' = 65 ('A'), 'J' = 74 ('J')
        row_letter := CHR(i);
        FOR col IN 1..12 LOOP
            INSERT INTO "HallSeats" ("SectorId", "Label", "Status")
            VALUES (standard_sector_id, row_letter || col, 0); -- 0 = Free
        END LOOP;
    END LOOP;

    -- VIP ряд: K — 6 місць
    FOR col IN 1..6 LOOP
        INSERT INTO "HallSeats" ("SectorId", "Label", "Status")
        VALUES (vip_sector_id, 'K' || col, 0); -- 0 = Free
    END LOOP;
END $$;

-- Частина 1: Заповнення таблиці SessionSeats для ВСІХ сеансів
-- Для кожного сеансу вставляємо всі сидіння з Red Hall (HallId = 1)
DO $$
DECLARE
    session_record RECORD;
    hall_seat_id BIGINT;
BEGIN
    -- Проходимося по кожному сеансу, який ми вставили
    FOR session_record IN SELECT "Id", "HallId" FROM "Sessions"
    LOOP
        -- Оскільки у нас лише одна зала (Red Hall, HallId = 1), ми знаємо, які сидіння брати.
        -- Вибираємо всі HallSeatId, які належать до секторів HallId = 1.
        FOR hall_seat_id IN SELECT hs."Id"
                             FROM "HallSeats" hs
                             JOIN "HallSectors" hsec ON hs."SectorId" = hsec."Id"
                             WHERE hsec."HallId" = 1 -- Завжди 1, оскільки у нас одна зала
        LOOP
            -- Вставляємо сидіння для поточного сеансу зі статусом "Вільне" (0)
            INSERT INTO "SessionSeats" ("SessionId", "HallSeatId", "Status")
            VALUES (session_record."Id", hall_seat_id, 0);
        END LOOP;
    END LOOP;
END $$;


-- Частина 2: Рандомізація статусу сидінь для ВСІХ сеансів
-- Робимо деякі місця зайнятими (Status = 1)
DO $$
DECLARE
    session_id BIGINT;
    seat_ids BIGINT[];
BEGIN
    -- Проходимося по всіх ID сеансів, які ми вставили (від 1 до 36)
    FOR session_id IN 1..36 LOOP -- Або SELECT "Id" FROM "Sessions"
        -- Отримуємо випадкові сидіння, пов’язані з поточним сеансом
        SELECT ARRAY(
            SELECT "HallSeatId"
            FROM "SessionSeats"
            WHERE "SessionId" = session_id
            ORDER BY random()
            LIMIT floor(random() * 21 + 10) -- Випадкова кількість від 10 до 30 місць
        )
        INTO seat_ids;

        -- Оновлюємо статус обраних сидінь на "Зайняте" (1)
        UPDATE "SessionSeats"
        SET "Status" = 1
        WHERE "SessionId" = session_id
          AND "HallSeatId" = ANY(seat_ids);
    END LOOP;
END $$;
